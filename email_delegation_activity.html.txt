        function selectOption(scene, option) {
            choices[`scene${scene}`] = option;
            
            const consequenceDiv = document.getElementById(`consequence${scene}`);
            const consequence = consequences[`scene${scene}`][option];
            
            consequenceDiv.className = `consequence show ${consequence.type}`;
            consequenceDiv.innerHTML = `
                <div class="consequence-title">Consequence:</div>
                <p>${consequence.text}</p>
            `;

            document.getElementById(`nav${scene}`).classList.remove('hidden');
            
            updateProgress();
        }

        function nextScene(sceneNum) {
            document.querySelectorAll('.scene').forEach(s => s.classList.remove('active'));
            document.getElementById(`scene${sceneNum}`).classList.add('active');
            window.scrollTo(0, 0);
        }

        function previousScene(sceneNum) {
            document.querySelectorAll('.scene').forEach(s => s.classList.remove('active'));
            document.getElementById(`scene${sceneNum}`).classList.add('active');
            window.scrollTo(0, 0);
        }

        function updateProgress() {
            const completed = Object.values(choices).filter(c => c > 0).length;
            const progress = (completed / 3) * 100;
            document.getElementById('progressBar').style.width = progress + '%';
        }

        function showSummary() {
            document.querySelectorAll('.scene').forEach(s => s.classList.remove('active'));
            document.getElementById('summary').classList.add('active');
            
            const goodChoices = Object.keys(choices).filter(key => {
                const scene = key.replace('scene', '');
                const choice = choices[key];
                return consequences[key][choice].type === 'good';
            }).length;

            const scoreText = goodChoices === 3 
                ? "üèÜ Excellent! You demonstrated strong AI delegation skills."
                : goodChoices === 2 
                ? "üëç Good job! You made mostly sound decisions."
                : "üìö Keep learning! Review the takeaways to improve your AI delegation strategy.";

            document.getElementById('scoreDisplay').textContent = scoreText;

            const choicesContent = document.getElementById('choicesContent');
            choicesContent.innerHTML = `
                <p><strong>Scene 1:</strong> You chose option ${choices.scene1}</p>
                <p><strong>Scene 2:</strong> You chose option ${choices.scene2}</p>
                <p><strong>Scene 3:</strong> You chose option ${choices.scene3}</p>
            `;

            window.scrollTo(0, 0);
        }

        function restart() {
            choices.scene1 = 0;
            choices.scene2 = 0;
            choices.scene3 = 0;
            
            document.querySelectorAll('.consequence').forEach(c => {
                c.classList.remove('show');

            document.querySelectorAll('.nav-buttons').forEach(n => n.classList.add('hidden'));
            
            document.querySelectorAll('.scene').forEach(s => s.classList.remove('active'));
            document.getElementById('scene1').classList.add('active');
            
            updateProgress();
            window.scrollTo(0, 0);
