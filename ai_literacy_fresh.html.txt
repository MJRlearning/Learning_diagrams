        function updateProgress() {
            const answeredQuestions = document.querySelectorAll('input[type="radio"]:checked').length;
            const totalQuestions = 10;
            const progress = (answeredQuestions / totalQuestions) * 100;
            document.getElementById('progressFill').style.width = progress + '%';
        }

        function handleOptionClick(event) {
            const option = event.currentTarget;
            const radio = option.querySelector('input[type="radio"]');
            const allOptions = option.parentNode.querySelectorAll('.option');
            
            allOptions.forEach(opt => opt.classList.remove('selected'));
            option.classList.add('selected');
            radio.checked = true;
            
            updateProgress();
        }

        function calculateResults() {
            const formData = new FormData(document.getElementById('assessmentForm'));
            const scores = {
                fundamentals: (parseInt(formData.get('q1')) + parseInt(formData.get('q2'))) / 2,
                ethics: (parseInt(formData.get('q3')) + parseInt(formData.get('q4'))) / 2,
                practical: (parseInt(formData.get('q5')) + parseInt(formData.get('q6'))) / 2,
                critical: (parseInt(formData.get('q7')) + parseInt(formData.get('q8'))) / 2,
                governance: (parseInt(formData.get('q9')) + parseInt(formData.get('q10'))) / 2
            };

            const overallScore = Object.values(scores).reduce((sum, score) => sum + score, 0) / 5;
            const level = Math.round(overallScore);

            return { scores, overallScore, level };
        }

        function displayResults(results) {
            const { scores, level } = results;
            
            document.getElementById('levelBadge').textContent = competencyLevels[level].name;
            document.getElementById('levelDescription').textContent = competencyLevels[level].description;

            const categoryScores = document.getElementById('categoryScores');
            categoryScores.innerHTML = `
                <div class="score-card">
                    <div class="score-value">${scores.fundamentals.toFixed(1)}/5</div>
                    <div>AI Fundamentals</div>
                </div>
                <div class="score-card">
                    <div class="score-value">${scores.ethics.toFixed(1)}/5</div>
                    <div>Ethics & Society</div>
                </div>
                <div class="score-card">
                    <div class="score-value">${scores.practical.toFixed(1)}/5</div>
                    <div>Practical Use</div>
                </div>
                <div class="score-card">
                    <div class="score-value">${scores.critical.toFixed(1)}/5</div>
                    <div>Critical Evaluation</div>
                </div>
                <div class="score-card">
                    <div class="score-value">${scores.governance.toFixed(1)}/5</div>
                    <div>Governance & Policy</div>
                </div>
            `;

            const recommendations = document.getElementById('recommendations');
            const plan = learningPlans[level];
            recommendations.innerHTML = plan.map(item => {
                const priorityClass = `priority-${item.priority.toLowerCase()}`;
                return `<div class="learning-item">
                    <a href="${item.url}" target="_blank">
                        <i data-lucide="${item.icon}" class="learning-item-icon"></i>
                        <span>
                            ${item.text}
                            <span class="priority-badge ${priorityClass}">${item.priority}</span>
                        </span>
                    </a>
                </div>`;
            }).join('');

            // Initialize Lucide icons
            setTimeout(() => {
                if (typeof lucide !== 'undefined') {
                    lucide.createIcons();
                }
            }, 100);

            document.getElementById('assessmentForm').style.display = 'none';
            document.getElementById('results').style.display = 'block';
        }

        function resetAssessment() {
            document.getElementById('assessmentForm').reset();
            document.getElementById('assessmentForm').style.display = 'block';
            document.getElementById('results').style.display = 'none';
            document.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));
            document.getElementById('progressFill').style.width = '0%';
            
            // Reinitialize icons after reset
            setTimeout(() => {
                if (typeof lucide !== 'undefined') {
                    lucide.createIcons();
                }
            }, 100);
        }

        document.addEventListener('DOMContentLoaded', function() {
            // Initialize Lucide icons with error handling
            setTimeout(() => {
                try {
                    if (typeof lucide !== 'undefined') {
                        lucide.createIcons();
                    }
                } catch (error) {
                    console.log('Icons loading...', error);
                }
            }, 500);
            
            document.querySelectorAll('.option').forEach(option => {
                option.addEventListener('click', handleOptionClick);
            });

            document.getElementById('assessmentForm').addEventListener('submit', function(e) {
                e.preventDefault();
                
                // Check if all questions are answered
                const answeredQuestions = document.querySelectorAll('input[type="radio"]:checked').length;
                
                if (answeredQuestions < 10) {
                    alert('Please answer all questions before completing the assessment.');
                    return;
                }
                
                const results = calculateResults();
                displayResults(results);
            });
        });
